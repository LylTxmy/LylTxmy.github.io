web 2

import re

import requests

url = ‘http://123.206.31.85:10002/'

r = requests.session()

text = r.get(url).text

calc = str(re.findall(“(.*?)

“, text))[2:-2]

ans = eval(calc)

data = {‘result’:ans}

res = r.post(url, data)

print(res.text)

使用脚本即可
